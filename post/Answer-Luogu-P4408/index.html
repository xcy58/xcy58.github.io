Luogu - P4408

正文

这题嘛... 也挺简单的? 比较水，首先树的直径求一波，然后遍历每个点与直径两端点的简单路径的长度，两条路径长度取个minminmin，然后把每个minminmin取个maxmaxmax。
然而是听了同机房神仙讲的。

为什么使用树的直径?

根据规则之一 Chris的父母总沿着两点间唯一的通路行走 而且 最坏情况下Chris的父母要耗费多长时间才能找到Chris ，我们显然可以知道，一个图上最长的两点的路径 (不可能会走不必要走的路径，因为题目说过 决定在尽量短的时间内 ) 必定是 树的直径 。
那么其中任意两点的距离就是 树的直径 ，我们这里假设为 A→BA \to BA→B 。

那么如何求出C点?

首先，我们先顺一下所要走的顺序为 C→A→BorC→B→AC \to A \to B \quad or \quad C \to B \to AC→A→BorC→B→A 。
也就是说， CCC 去的地方一定是 AAA 或 BBB 中距离 CCC 最短的点。既然要最费时，那么就要求出最短中的最大，才能得出最佳答案。
那么！我们只需要对 AAA 和 BBB 跑一遍最长路， 然后，枚举一下 CCC 点，对每个 min⁡(len(C→A),len(C→B))\min(len(C\to A),len(C\to B))min(len(C→A),len(C→B)) 取一遍 max⁡\maxmax 就好了，C点就是求出最大长度的那个点啦。

所以答案是什么?

好！答案就是 直径的长度+C点与A或B点的长度中最短的长度 。
最后！首尾呼应！Use "long long"！Use "long long"！Use "long long"！

using namespace std;
#define reint register int
#define ull unsigned ll
#define INF 0x3f3f3f3f
#define ll long long
#define GC getchar()
#define R read()
ll read(){
    char c=GC; ll s=0,f=1;
    while(c<'0'||c>'9'){if(c=='-')f=-f;c=GC;}
    while(c>='0'&&c<='9'){s=s*10+c-'0';c=GC;}
    return s*f;
}
struct node { ll to,v; };
node make_node(ll to,ll v) { node t; t.to=to; t.v=v; return t; }
ll n,m,pluslen;
vector<node> e[200005],dij[200005];
ll a,b,len,deep[200005];
struct Dijkstra{
    struct nodeDis{
        ll len,v;
        bool operator < (const nodeDis &Get) const {
            return len>Get.len;
        }
    };
    nodeDis make_nodeDis(ll len,ll v){
        nodeDis t; t.len=len; t.v=v;
        return t;
    }
    priority_queue<nodeDis> q;
    ll vis[200005],dis[200005];
    void dijkstra(ll XD){
        while(!q.empty()) q.pop();
        for(ll i=1;i<=n;++i) vis[i]=0;
        for(ll i=1;i<=n;++i) dis[i]=INF;
        dis[XD]=0; q.push(make_nodeDis(0,XD));
        while(!q.empty()){
            ll u=q.top().v; q.pop();
            if(vis[u]) continue;
            vis[u]=1;
            for(ll i=0;i<dij[u].size();++i){
                ll v=dij[u][i].to,w=dij[u][i].v;
                if(dis[v]>dis[u]+w&&!vis[v]){
                    dis[v]=dis[u]+w;
                    q.push(make_nodeDis(dis[v],v));
                }
            }
        }
    }
}x,y;
//以上是奇怪的dijkstra
void dfs(ll u,ll fa){
    if(len<deep[u]) { len=deep[u]; a=u; }
    for(ll i=0,uLen=e[u].size();i<uLen;++i){
        ll v=e[u][i].to,w=e[u][i].v;
        if(v==fa) continue;
        deep[v]=deep[u]+w; dfs(v,u);
    }
}
//以上是用来求直径的dfs
int main(){
    n=R; m=R;
    for(ll i=1;i<=m;++i){
        ll u=R,v=R,w=R;
        e[u].push_back(make_node(v,w));
        e[v].push_back(make_node(u,w));
        //为求直径而建图
        dij[u].push_back(make_node(v,-w));
        dij[v].push_back(make_node(u,-w));
        //为跑 dijkstra 而建图
    }
    dfs(1,0);
    memset(deep,0,sizeof(deep)); len=0; b=a;
    dfs(b,0);
    //求直径
    x.dijkstra(a); y.dijkstra(b);
    for(ll i=1;i<=n;++i)
        pluslen=max(pluslen,min(-x.dis[i],-y.dis[i]));
    //最短路&求C点(这里直接求的是C到A或B的最短距离)
    printf("%lld",len+pluslen);
    return 0;
}
